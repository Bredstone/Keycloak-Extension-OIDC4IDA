.[] | {
    verified_claims: {
      verification: {
        trust_framework: .[] | select(.verification.trust_framework != "ida") | .verification.trust_framework,
        time: .verification.time,
        evidence: [(.verification.evidence[] | select(.type == "document") | { type: .type, method: .method, document_details: { type: .document_details.type } })]
      },
      claims: {
        given_name: .claims.given_name,
        family_name: .claims.family_name,
        birthdate: .claims.birthdate
      }
    }
  }

.verified_claims.verification.time | (try(. | strptime("%Y-%m-%dT%H:%MZ")) // try(. | strptime("%Y-%m-%dT%H:%M:%SZ"))) | mktime
